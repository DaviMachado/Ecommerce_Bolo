create database ecommerce;

drop table cliente;

create table cliente (
	id	BIGINT	NOT	NULL	AUTO_INCREMENT,
	login		VARCHAR(50),
	senha		VARCHAR(50),
	nome		VARCHAR(50),
	cpf		VARCHAR(14),
	dt_nasc 	DATE,
	cd_cliente	VARCHAR(30),
	fl_ativo NUMERIC(1),
	primary	key	(id)
);

use ecommerce;

select * from cliente;
